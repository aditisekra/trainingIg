            var person = {
                details: function (city, state, country) {
                  var name = this.fname + " " + this.lname;
                  return (
                    "Name: " +
                    name +
                    " City:  " +
                    city +
                    "State: " +
                    state +
                    "Country: " +
                    country
                  );
                },
              };
        
              var obj = {
                fname: prompt("Enter your First Name "),
                lname: prompt("Enter your Last Name "),
              };
              var city = prompt("Enter your City ");
              var state = prompt("Enter your State ");
              var country = prompt("Enter your Country ");
              var arr = [city, state, country];
              //using call and apply
              document.body.innerHTML =
                " Using call" +
                person.details.call(obj, city, state, country) +
                "Using apply" +
                person.details.apply(obj, arr) +"";
